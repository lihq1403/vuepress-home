# 很高兴认识你
```php
<?php

class Me {
    public function find() {
        return true;
    }

    public function met() {
        return mt_rand(1, 10000);
    }
}

class Your
{
    public function self() {
        return 1;
    }
}

$me = new Me();
$your = new Your();

while ($me->find()) {
    if ($me->met() == $your->self()) {
        print ("Hello,World!");
        break;
    }
}

```
